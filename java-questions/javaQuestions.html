Strings

  • Method that checks if string is a Palindrome.
    public static boolean isPalindrome(String s) {
    int inputLength = s.length();

    if (!(inputLength > 0 && s instanceof String && s != null)) {
        return false;
    }
    for (int i = 0; i < inputLength / 2; i += 1) {
        if (s.toLowerCase().charAt(i) != s.toLowerCase().charAt(inputLength - 1 - i)) {
          return false;
        }
    }
    return true;
}
  • Method that checks if string1 is a rotation of string2.
    public static boolean isRotated(String s1, String s2) {
    if (s1 == null || s2 == null) {
        return false;
    }
    return (s1.length() == s2.length() && s1.concat(s1).contains(s2));
}
  • Method that reverses a string.
    public static String reverseString(String s) {
    if (s == null) {
        throw new IllegalArgumentException("can't be null");
    }
    StringBuilder sb = new StringBuilder(s);
    return sb.reverse().toString();
}
